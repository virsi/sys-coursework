 610  ;=== Функция, которая в зависимости от флага cursiveEnabled меняет начертание символа с курсива на обычное и наоброт
 611  ;=== Сама смена происходит в процедуре changeFont, а здесь подготавливаются данные
 612  setCursive proc
 613  	push ES ; сохраняем регистры
 614  	push AX
 615  	push CS
 616  	pop ES
 617  
 618  	cmp cursiveEnabled, true
 619  	jne _restoreSymbol
 620  	; если флаг равен true, выполняем замену символа на курсивный вариант,
 621  	; предварительно сохраняя старый символ в savedSymbol
 622  
 623  	call saveFont
 624  	mov CL, charToCursiveIndex
 625  _shifTtable:
 626  	; мы получаем в BP таблицу всех символов. адрес указывает на символ 0
 627  	; поэтому нуэно совершить сдвиг 16*X - где X - код символа
 628  	add BP, 16
 629  	loop _shiftTable
 630  
 631  	; пpи savefont смещается pегистp ES
 632  	; поэтомy пpиходится делать такие махинации, чтобы
 633  	; записать полyченный элемент в savedSymbol
 634  	; swap(ES, DS) и сохранение старого значения DS
 635  	push DS
 636  	pop AX
 637  	push ES
 638  	pop DS
 639  	push AX
 640  	pop ES
 641  	push AX
 642  
 643  	mov SI, BP
 644  	lea DI, savedSymbol
 645  	; сохpаняем в пеpеменнyю savedSymbol
 646  	; таблицy нyжного символа
 647  	mov CX, 16
 648  	; movsb из DS:SI в ES:DI
 649  	rep movsb
 650  	; исходные позиции сегментов возвpащены
 651  	pop DS ; восстановление DS
 652  
 653  	; заменим написание символа на кypсив
 654  	mov CX, 1
 655  	mov DH, 0
 656  	mov DL, charToCursiveIndex
 657  	lea BP, cursiveSymbol
 658  	call changeFont
 659  	jmp _exitSetCursive
 660

 662  	; если флаг равен 0, выполняем замену курсивного символа на старый вариант
 663  
 664  	mov CX, 1
 665  	mov DH, 0
 666  	mov DL, charToCursiveIndex
 667  	lea bp, savedSymbol
 668  	call changeFont

 671  	pop AX
 672  	pop ES
 673  	ret

 676  ;=== Функция смены начертания символа (курсив/нормальное)
 677  ;===
 678  ; *** входные данные
 679  ; DL = номер символа для замены
 680  ; CX = Кол-во символов заменяемых изображений символов
 681  ; (начиная с символа указанного в DX)
 682  ; ES:bp = адрес таблицы
 683  ;
 684  ; *** описание работы процедуры
 685  ; Происходит вызов int 10h (видеосервис)
 686  ; с функцией AH = 11h (функции знакогенератора)
 687  ; Параметр AL = 0 сообщает, что будет заменено изображение
 688  ; символа для текущего шрифта
 689  ; В случаях, когда AL = 1 или 2, будет заменено изображение
 690  ; только для опредленного шрифта (8x14 и 8x8 соответственно)
 691  ; Параметр BH = 0Eh сообщает, что на опредление каждого изображения символа
 692  ; расходуется по 14 байт (режим 8x14 бит как раз 14 байт)
 693  ; Параметр BL = 0 - блок шрифта для загрузки (от 0 до 4)
 694  ;
 695  ; *** результат
 696  ; изображение указанного(ых) символа(ов) будет заменено
 697  ; на предложенное пользователем.
 698  ; Изменению подвергнутся все символы, находящиеся на экране,
 699  ; то есть если изображение заменено, старый вариант нигде уже не проявится
